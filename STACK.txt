STACK

#include<stdio.h>
int stack[10];
int isFull(int t1){
    if(t1==9)
        return(0);
    else
        return(1);
    }
   
int isEmpty(int t2){
    if(t2==-1)
        return(0);
    else
        return(1);
}
   
int push(int t){
        int x,flag;
        flag=isFull(t);
        isEmpty(t);
        if(flag==0){
            printf("Stack Overflow");
            return(t);
        }
        else{
            printf("Enter x:- ");
            scanf("%d",&x);
            t=t+1;
            stack[t]=x;
            return(t);
        }
}
int pop(int y){
    int flag,x;
    flag=isEmpty(y);
        if(flag==0){
            printf("Stack Underflow");
            printf("\n%d",y);
            return(-1);
            }
        else{
            x=stack[y];
            y=y-1;
            printf("Element is popped");
            printf("\n%d",x);
            return y;
        }
}

void main(){
    int top=-1,ch;
    do{
        printf("\nEnter 1.push 2.pop 3.exit:-");
        scanf("%d",&ch);
        switch(ch){
            case 1:
                top=push(top);
                break;
            case 2:
                top=pop(top);
                break;
            case 3:
                printf("End of Program");
                break;
            default:
                printf("Error");
        }
    }while(ch!=3);
}
